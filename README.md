## Install

```bash
npm install --save react-twitter-search-component
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-twitter-search-component'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
